package com.silence.quartz.utils;

import com.silence.common.exception.UtilException;
import org.quartz.CronExpression;

import java.text.ParseException;
import java.util.Date;

/**
 * @author SilenceLambs
 * @apiNote cron表达式工具类
 */
public class CronUtils {

  /**
   * 验证Cron表达式是否有效。
   *
   * @param cronExpression 待验证的Cron表达式。
   * @throws UtilException 如果Cron表达式无效，则抛出UtilException异常。
   */
  public static void validateCron(String cronExpression) {
    try {
      CronExpression.isValidExpression(cronExpression); // 验证Cron表达式的有效性
    } catch (Exception e) {
      throw new UtilException("表达式无效"); // 如果表达式无效，抛出异常
    }
  }

  /**
   * 根据给定的Cron表达式计算下一个执行时间
   *
   * @param cronExpression Cron表达式，用于定义时间触发规则
   * @return 计算得到的下一个执行时间的Date对象
   * @throws UtilException 如果Cron表达式解析失败，则抛出UtilException异常
   */
  public static Date getNextExecution(String cronExpression) {
    try {
      // 解析Cron表达式
      CronExpression cron = new CronExpression(cronExpression);
      // 计算从当前时间开始的下一个执行时间
      return cron.getNextValidTimeAfter(new Date(System.currentTimeMillis()));
    } catch (ParseException e) {
      // 将解析异常封装并抛出
      throw new UtilException(e.getMessage());
    }
  }
}

  • 定时任务工具🌳 ScheduleUtils.java
java
package com.silence.quartz.utils;

import com.silence.common.exception.ServiceException;
import com.silence.common.utils.StringUtils;
import com.silence.quartz.domain.JobEntity;
import com.silence.quartz.enums.StatusEnums;
import com.silence.quartz.exception.TaskException;
import com.silence.quartz.exception.TaskException.Code;
import com.silence.quartz.handle.ConcurrentHandler;
import com.silence.quartz.handle.UnConcurrentHandler;
import org.quartz.*;

import static com.silence.quartz.constants.ScheduleConstants.*;

/**
 * @author SilenceLamb
 * @apiNote 定时任务工具
 */
public class ScheduleUtils {
  /**
   * 创建定时任务
   *
   * @param scheduler Quartz调度器，用于管理定时任务
   * @param jobEntity 任务实体，包含任务的详细信息，如执行类、执行时间表达式等
   * @throws SchedulerException 调度器操作异常
   * @throws ServiceException   服务层异常
   * @throws TaskException      任务层异常
   */
  public static void createScheduleTask(Scheduler scheduler, JobEntity jobEntity) throws SchedulerException, ServiceException, TaskException {
    /* 获取任务对应的Job类 */
    Class<? extends Job> jobClass = getQuartzJobClass(jobEntity);

    /* 构建任务详情，包括任务ID和分组 */
    Long jobId = jobEntity.getJobId();
    String jobGroup = jobEntity.getJobGroup();
    /*  创建一个新的JobDetail实例。这个方法通过指定的jobClass和标识符（包括jobId和jobGroup）来构建一个JobDetail对象 */
    JobDetail jobDetail = JobBuilder.newJob(jobClass).withIdentity(getJobKey(jobId, jobGroup)).build();

    /* 任务 Cron表达式 */
    String cronExpression = jobEntity.getCronExpression();
    /*  构建任务的触发器，使用Cron表达式定义任务的执行时间 */
    CronScheduleBuilder cronScheduleBuilder = CronScheduleBuilder.cronSchedule(cronExpression);
    /* 设置定时任务策略 */
    cronScheduleBuilder = handleCronScheduleMisfirePolicy(jobEntity, cronScheduleBuilder);

    /* 创建触发器，指定触发器的ID和调度表达式 */
    CronTrigger trigger = TriggerBuilder.newTrigger().withIdentity(getTriggerKey(jobId, jobGroup)).withSchedule(cronScheduleBuilder).build();

    /* 将任务实体作为参数放入JobDetail中，以便在任务执行时访问 */
    jobDetail.getJobDataMap().put(TASK_PROPERTIES, jobEntity);

    /* 检查是否已存在同名任务，若存在则先删除后重新创建 */
    if (scheduler.checkExists(getJobKey(jobId, jobGroup))) {
      scheduler.deleteJob(getJobKey(jobId, jobGroup));
    }

    /* 判断任务是否有效，若有效则调度任务 */
    if (StringUtils.isNotNull(CronUtils.getNextExecution(jobEntity.getCronExpression()))) {
      scheduler.scheduleJob(jobDetail, trigger);
    }

    // 若任务状态为暂停，则暂停任务
    if (jobEntity.getStatus().equals(StatusEnums.NORMAL.getValue())) {
      scheduler.pauseJob(ScheduleUtils.getJobKey(jobId, jobGroup));
    }
  }


  /**
   * 根据JobEntity的并发属性获取对应的Quartz作业类。
   *
   * @param jobEntity 作业实体，包含作业的详细配置信息，其中的并发属性将决定返回哪个作业处理类。
   * @return 返回一个作业处理类，如果作业允许并发执行，则返回ConcurrentHandler类；否则返回UnConcurrentHandler类。
   */
  private static Class<? extends Job> getQuartzJobClass(JobEntity jobEntity) {
    // 判断作业是否允许并发执行
    boolean isConcurrent = StatusEnums.CONCURRENT.getValue().equals(jobEntity.getConcurrent());
    // 根据并发属性决定返回的任务处理类
    return isConcurrent ? ConcurrentHandler.class : UnConcurrentHandler.class;
  }


  /**
   * 获取触发器的键。
   *
   * @param jobId    作业的ID，用于构造触发器的名称。
   * @param jobGroup 作业所属的组别，作为触发器的组标识。
   * @return 返回根据作业ID和组别构造的触发器键。
   */
  public static TriggerKey getTriggerKey(Long jobId, String jobGroup) {
    // 通过作业ID和组别创建触发器键
    return TriggerKey.triggerKey(TASK_CLASS_NAME + jobId, jobGroup);
  }


  /**
   * 获取作业的键。
   *
   * @param jobId    作业的唯一标识符。
   * @param jobGroup 作业所属的组名。
   * @return 返回一个作业键（JobKey），由任务类名和作业ID拼接而成，归属指定的作业组。
   */
  public static JobKey getJobKey(Long jobId, String jobGroup) {
    // 根据任务ID和组名创建JobKey
    return JobKey.jobKey(TASK_CLASS_NAME + jobId, jobGroup);
  }


  /**
   * 设置定时任务策略。
   *
   * @param jobEntity 作业实体，包含误触发策略信息。
   * @param cb        CronScheduleBuilder对象，用于构建Cron定时策略。
   * @return 根据误触发策略调整后的CronScheduleBuilder对象。
   * @throws TaskException 如果作业实体的误触发策略不支持，则抛出任务异常。
   */
  public static CronScheduleBuilder handleCronScheduleMisfirePolicy(JobEntity jobEntity, CronScheduleBuilder cb) throws TaskException {
    switch (jobEntity.getMisfirePolicy()) {
      case MISFIRE_DEFAULT:
        // 默认策略，直接返回原有的CronScheduleBuilder对象
        return cb;
      case MISFIRE_IGNORE_MISFIRES:
        // 忽略误触发，返回设置为忽略误触发的CronScheduleBuilder对象
        return cb.withMisfireHandlingInstructionIgnoreMisfires();
      case MISFIRE_FIRE_AND_PROCEED:
        // 立即执行，返回设置为立即执行的CronScheduleBuilder对象
        return cb.withMisfireHandlingInstructionFireAndProceed();
      case MISFIRE_DO_NOTHING:
        // 不做任何处理，返回设置为不处理的CronScheduleBuilder对象
        return cb.withMisfireHandlingInstructionDoNothing();
      default:
        // 如果策略无效，则抛出异常
        throw new TaskException("The task misfire policy '" + jobEntity.getMisfirePolicy()
          + "' cannot be used in cron schedule tasks", Code.CONFIG_ERROR);
    }
  }
}

java
package com.silence.quartz.utils;

import com.silence.common.utils.SpringUtils;
import com.silence.common.utils.StringUtils;
import com.silence.quartz.domain.JobEntity;

import java.lang.reflect.InvocationTargetException;
import java.lang.reflect.Method;
import java.util.LinkedList;
import java.util.List;

/**
 * @author SilenceLambs
 * @apiNote 任务执行工具
 */
public class TaskInvokeUtils {
  public static void invokeMethod(JobEntity jobEntity) throws Exception {
    String invokeTarget = jobEntity.getInvokeTarget();
    String beanName = getBeanName(invokeTarget);
    String methodName = getMethodName(invokeTarget);
    List<Object[]> methodParams = getMethodParams(invokeTarget);

    if (!isValidClassName(beanName)) {
      Object bean = SpringUtils.getBean(beanName);
      invokeMethod(bean, methodName, methodParams);
    } else {
      Object bean = Class.forName(beanName).getDeclaredConstructor().newInstance();
      invokeMethod(bean, methodName, methodParams);
    }
  }

  /**
   * 调用任务方法
   *
   * @param bean         目标对象
   * @param methodName   方法名称
   * @param methodParams 方法参数
   */
  private static void invokeMethod(Object bean, String methodName, List<Object[]> methodParams)
    throws NoSuchMethodException, SecurityException, IllegalAccessException, IllegalArgumentException,
    InvocationTargetException {
    if (StringUtils.isNotNull(methodParams) && methodParams.size() > 0) {
      Method method = bean.getClass().getMethod(methodName, getMethodParamsType(methodParams));
      method.invoke(bean, getMethodParamsValue(methodParams));
    } else {
      Method method = bean.getClass().getMethod(methodName);
      method.invoke(bean);
    }
  }

  /**
   * 校验是否为为class包名
   *
   * @param invokeTarget 名称
   * @return true是 false否
   */
  public static boolean isValidClassName(String invokeTarget) {
    return StringUtils.countMatches(invokeTarget, ".") > 1;
  }

  /**
   * 获取bean名称
   *
   * @param invokeTarget 目标字符串
   * @return bean名称
   */
  public static String getBeanName(String invokeTarget) {
    String beanName = StringUtils.substringBefore(invokeTarget, "(");
    return StringUtils.substringBeforeLast(beanName, ".");
  }

  /**
   * 获取bean方法
   *
   * @param invokeTarget 目标字符串
   * @return method方法
   */
  public static String getMethodName(String invokeTarget) {
    String methodName = StringUtils.substringBefore(invokeTarget, "(");
    return StringUtils.substringAfterLast(methodName, ".");
  }

  /**
   * 获取method方法参数相关列表
   *
   * @param invokeTarget 目标字符串
   * @return method方法相关参数列表
   */
  public static List<Object[]> getMethodParams(String invokeTarget) {
    String methodStr = StringUtils.substringBetween(invokeTarget, "(", ")");
    if (StringUtils.isEmpty(methodStr)) {
      return null;
    }
    String[] methodParams = methodStr.split(",(?=([^\"']*[\"'][^\"']*[\"'])*[^\"']*$)");
    List<Object[]> classs = new LinkedList<>();
    for (String methodParam : methodParams) {
      String str = StringUtils.trimToEmpty(methodParam);
      // String字符串类型，以'或"开头
      if (StringUtils.startsWithAny(str, "'", "\"")) {
        classs.add(new Object[]{StringUtils.substring(str, 1, str.length() - 1), String.class});
      }
      // boolean布尔类型，等于true或者false
      else if ("true".equalsIgnoreCase(str) || "false".equalsIgnoreCase(str)) {
        classs.add(new Object[]{Boolean.valueOf(str), Boolean.class});
      }
      // long长整形，以L结尾
      else if (StringUtils.endsWith(str, "L")) {
        classs.add(new Object[]{Long.valueOf(StringUtils.substring(str, 0, str.length() - 1)), Long.class});
      }
      // double浮点类型，以D结尾
      else if (StringUtils.endsWith(str, "D")) {
        classs.add(new Object[]{Double.valueOf(StringUtils.substring(str, 0, str.length() - 1)), Double.class});
      }
      // 其他类型归类为整形
      else {
        classs.add(new Object[]{Integer.valueOf(str), Integer.class});
      }
    }
    return classs;
  }

  /**
   * 获取参数类型
   *
   * @param methodParams 参数相关列表
   * @return 参数类型列表
   */
  public static Class<?>[] getMethodParamsType(List<Object[]> methodParams) {
    Class<?>[] classs = new Class<?>[methodParams.size()];
    int index = 0;
    for (Object[] os : methodParams) {
      classs[index] = (Class<?>) os[1];
      index++;
    }
    return classs;
  }

  /**
   * 获取参数值
   *
   * @param methodParams 参数相关列表
   * @return 参数值列表
   */
  public static Object[] getMethodParamsValue(List<Object[]> methodParams) {
    Object[] objectClass = new Object[methodParams.size()];
    int index = 0;
    for (Object[] os : methodParams) {
      objectClass[index] = (Object) os[0];
      index++;
    }
    return objectClass;
  }
}
