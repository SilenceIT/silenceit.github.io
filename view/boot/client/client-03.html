package com.silence.framework.config;

import com.baomidou.mybatisplus.core.config.GlobalConfig;
import com.silence.framework.hanlder.MybatisHandler;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.core.io.ClassPathResource;
import org.springframework.jdbc.datasource.init.DataSourceInitializer;
import org.springframework.jdbc.datasource.init.ResourceDatabasePopulator;
import org.springframework.transaction.annotation.EnableTransactionManagement;

import javax.annotation.Resource;
import javax.sql.DataSource;

/**
 * @author SilenceLamb
 * @apiNote Mybatis配置类
 */
@Configuration
@EnableTransactionManagement
public class MybatisConfig {
    private static final Logger log = LoggerFactory.getLogger(MybatisConfig.class);

    @Resource
    private MybatisHandler mybatisHandler; // MybatisPlus处理器

    /**
     * 配置全局设置
     *
     * @return 返回配置好的GlobalConfig对象
     */
    @Bean("globalConfig")
    public GlobalConfig globalConfig() {
        GlobalConfig globalConfig = new GlobalConfig();
        // 禁用启动时的banner输出，以保证启动过程简洁
        globalConfig.setBanner(false);
        // 设置MetaObjectHandler
        globalConfig.setMetaObjectHandler(mybatisHandler);
        return globalConfig;
    }

    /**
     * 初始化数据库数据源。
     * 这个方法创建一个数据源初始化器（DataSourceInitializer），它负责在应用启动时根据预定义的SQL脚本对数据库进行初始化。
     *
     * @return DataSourceInitializer 数据源初始化器，配置了数据库连接源和数据库填充对象。
     */
    @Bean
    public DataSourceInitializer dataSourceInitializer() {
        log.info("正在初始化数据库:{}...","silence-boot");
        // 创建资源数据库填充器，并添加SQL脚本资源
        ResourceDatabasePopulator resourceDatabasePopulator = new ResourceDatabasePopulator();
        resourceDatabasePopulator.addScript(new ClassPathResource("sql/silence-boot.sql"));
        resourceDatabasePopulator.setContinueOnError(true); // 设置为遇到错误继续执行

        // 配置数据源初始化器
        DataSourceInitializer initializer = new DataSourceInitializer();
        initializer.setDataSource(dataSource); // 设置数据源
        initializer.setDatabasePopulator(resourceDatabasePopulator); // 设置数据库填充器

        log.info("数据库:{}初始化完成","silence-boot");
        return initializer;
    }


    @Resource
    private DataSource dataSource;
}
  • Mybatis配置类🌳 MapperConfig.java
java
package com.silence.framework.config;

import com.baomidou.mybatisplus.core.config.GlobalConfig;
import com.baomidou.mybatisplus.extension.spring.MybatisSqlSessionFactoryBean;
import com.silence.common.properties.MybatisProperties;
import com.silence.common.utils.PathUtils;
import org.apache.ibatis.io.VFS;
import org.apache.ibatis.session.SqlSessionFactory;
import org.mybatis.spring.boot.autoconfigure.SpringBootVFS;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.core.io.DefaultResourceLoader;
import org.springframework.transaction.annotation.EnableTransactionManagement;

import javax.annotation.Resource;
import javax.sql.DataSource;

/**
 * @author SilenceLamb
 * @apiNote 支持*扫描
 */
@Configuration
@EnableTransactionManagement
public class MapperConfig {
  @Resource
  private MybatisProperties mybatisProperties;
  @Resource
  private GlobalConfig globalConfig;

  @Bean
  public SqlSessionFactory sqlSessionFactory(DataSource dataSource) throws Exception {
    /* 将指定的类添加到VFS实现列表中 */
    VFS.addImplClass(SpringBootVFS.class);
    /* 依赖Spring容器完成MyBatis的SqlSessionFactory对象的创建 */
    final MybatisSqlSessionFactoryBean sessionFactoryBean = new MybatisSqlSessionFactoryBean();
    /* 配置数据源 */
    sessionFactoryBean.setDataSource(dataSource);
    /* 配置需要定义别名的实体类的包 */
    sessionFactoryBean.setTypeAliasesPackage(PathUtils.getClassPackage(mybatisProperties.getTypeAliasesPackage(), "/*.class"));
    /* 配置mapper文件的路径 */
    sessionFactoryBean.setMapperLocations(PathUtils.getFilePath(mybatisProperties.getMapperLocations()));
    /* 设置配置文件位置 */
    sessionFactoryBean.setConfigLocation(new DefaultResourceLoader().getResource(mybatisProperties.getConfigLocation()));
    /* 设置全局配置到SessionFactoryBean中 */
    sessionFactoryBean.setGlobalConfig(globalConfig);
    return sessionFactoryBean.getObject();
  }
}
