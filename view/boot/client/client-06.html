package com.silence.common.utils;

import com.silence.common.utils.string.StringUtils;
import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;

import javax.servlet.http.HttpServletRequest;
import java.util.Date;
import java.util.Map;

/**
 * @author SilenceLambs
 * @apiNote 用于生成JSON Web Token (JWT)的工具类
 */
public class JUtils {

  /**
   * 创建并返回一个JWT（JSON Web Token）字符串
   *
   * @param claims     JWT中包含的声明信息
   * @param algorithm  签名算法，用于确保JWT的完整性和防篡改
   * @param secret     用于签名的密钥
   * @return 生成的JWT字符串
   */
  public static String createToken(Map<String, Object> claims, SignatureAlgorithm algorithm, String secret) {
    return Jwts.builder()
      .setClaims(claims)
      .setNotBefore(new Date())
      .signWith(algorithm, secret)
      .compact();
  }

  /**
   * 创建并返回一个JWT（JSON Web Token）字符串
   *
   * @param claims     JWT中包含的声明信息
   * @param expireTime JWT的过期时间，单位为毫秒
   * @param algorithm  签名算法，用于确保JWT的完整性和防篡改
   * @param secret     用于签名的密钥
   * @return 生成的JWT字符串
   */
  public static String createToken(Map<String, Object> claims, long expireTime, SignatureAlgorithm algorithm, String secret) {
    return Jwts.builder()
      .setClaims(claims)
      .setNotBefore(new Date())
      .setExpiration(new Date(System.currentTimeMillis() + expireTime))
      .signWith(algorithm, secret)
      .compact();
  }

  /**
   * 解析JWT令牌并返回其中的声明（Claims）
   *
   * @param token  待解析的JWT令牌字符串
   * @param secret 用于签名JWT令牌的密钥
   * @return 解析得到的Claims对象，包含JWT中的声明信息
   */
  public static Claims parseToken(String token, String secret) {
    // 使用JWT解析器并设置签名密钥，然后解析令牌为Claims对象
    return Jwts.parser() // 创建JWT解析器
      .setSigningKey(secret) // 设置用于验证签名的密钥
      .parseClaimsJws(token) // 解析JWT令牌为ClaimsJws对象
      .getBody(); // 获取JWT令牌的主体（声明体）

  }

  /**
   * 从HTTP请求中提取请求令牌
   *
   * @param authorization 指定请求头的名称，通常用于授权信息的传递。
   * @param tokenPrefix   期望令牌以该前缀开头，如果匹配则移除该前缀。
   * @param request       HttpServletRequest对象，用于获取请求头信息。
   * @return 返回提取到的令牌字符串，如果没有匹配的令牌，则返回null或原请求头中的令牌值。
   */
  public static String getRequestToken(String authorization, String tokenPrefix, HttpServletRequest request) {
    String token = request.getHeader(authorization);
    if (StringUtils.isNotEmpty(token) && token.startsWith(tokenPrefix)) {
      return token.replace(tokenPrefix, "");
    }
    return token;
  }

  /**
   * 判断令牌（token）是否过期
   *
   * @param token  用于验证的JWT令牌字符串
   * @param secret 用于签名JWT令牌的密钥
   * @return 如果令牌已过期，则返回true；否则返回false
   */
  public static boolean isExpiration(String token, String secret) {
    // 使用JWT解析器和密钥解析令牌，并提取声明
    Claims claims = Jwts.parser().setSigningKey(secret).parseClaimsJws(token).getBody();
    // 判断令牌的过期时间是否早于当前时间
    return claims.getExpiration().before(new Date());
  }
}

package com.silence.framework.security.context;

import com.silence.common.constants.StatusConstants;
import com.silence.common.pojo.exception.ServiceException;
import com.silence.common.pojo.base.LoginUser;
import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.context.SecurityContextHolder;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;


/**
 * @author SilenceLambs
 * @apiNote 安全认证工具
 */
public class SecurityContext {
  /**
   * 获取当前登录的用户信息
   * 该方法尝试从认证信息中提取登录用户的信息
   *
   * @return LoginUser 返回登录用户的信息对象
   * @throws ServiceException 如果获取用户信息过程中发生异常，则抛出服务异常
   */
  public static LoginUser getLoginUser() {
    try {
      // 尝试从认证信息中获取当前登录用户的信息
      return (LoginUser) getAuthentication().getPrincipal();
    } catch (Exception e) {
      // 如果获取过程中出现异常，抛出服务异常
      throw new ServiceException(StatusConstants.UNAUTHORIZED, "获取用户信息异常");
    }
  }

  /**
   * 获取当前的认证信息
   *
   * @return 返回当前的Authentication对象，代表当前用户的身份认证信息
   */
  public static Authentication getAuthentication() {
    return SecurityContextHolder.getContext().getAuthentication();
  }

  /**
   * 设置认证信息
   *
   * @param authentication 要设置的UsernamePasswordAuthenticationToken对象，包含用户的认证信息
   *                       用于设置当前线程的安全上下文
   */
  public static void setAuthentication(UsernamePasswordAuthenticationToken authentication) {
    SecurityContextHolder.getContext().setAuthentication(authentication);
  }

  /**
   * 检查输入的密码是否与经过BCrypt加密的密码匹配。
   *
   * @param inputPassWord   用户输入的原始密码。
   * @param encodedPassword 经过BCrypt加密的密码字符串。
   * @return 返回一个布尔值，如果输入的密码与加密后的密码匹配，则返回true；否则返回false。
   */
  public static boolean matchesPassword(String inputPassWord, String encodedPassword) {
    BCryptPasswordEncoder passwordEncoder = new BCryptPasswordEncoder();
    // 使用BCryptPasswordEncoder的matches方法验证密码是否匹配
    return passwordEncoder.matches(inputPassWord, encodedPassword);
  }

}

package com.silence.framework.security.context;

import org.springframework.security.core.Authentication;

/**
 * @author SilenceLamb
 * @apiNote 身份验证信息线程安全存储类
 */
public class AuthenticationContext {
  // 使用ThreadLocal来存储Authentication对象，确保线程安全
  private static final ThreadLocal<Authentication> context = new ThreadLocal<>();

  /**
   * 从当前线程中获取认证信息。
   * @return Authentication 当前线程的认证信息。
   * @throws IllegalStateException 如果在当前线程中没有设置认证信息。
   */
  public static Authentication getAuthentication() {
    return context.get();
  }

  /**
   * 将认证信息存储到当前线程中。
   * @param authentication 需要存储的认证信息。
   */
  public static void setAuthentication(Authentication authentication) {
    context.set(authentication);
  }

  /**
   * 从当前线程中移除认证信息。
   */
  public static void removeAuthentication() {
    context.remove();
  }
}

package com.silence.framework.security.service;

import cn.hutool.core.util.IdUtil;
import com.silence.common.constants.LoginConstants;
import com.silence.common.core.redis.RedisCache;
import com.silence.common.core.redis.RedisCacheKey;
import com.silence.common.exception.ServiceException;
import com.silence.common.properties.TokenProperties;
import com.silence.common.utils.JwtUtils;
import com.silence.common.utils.string.StringUtils;
import com.silence.server.domain.model.LoginUser;
import io.jsonwebtoken.Claims;
import io.jsonwebtoken.SignatureAlgorithm;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;
import javax.servlet.http.HttpServletRequest;
import java.util.Date;
import java.util.HashMap;
import java.util.concurrent.TimeUnit;

/**
 * Token服务接口实现类
 * 提供用户登录Token的创建、刷新、验证等相关操作
 */
@Component
public class TokenService {
    private static final Logger LOGGER = LoggerFactory.getLogger(TokenService.class);

    @Resource
    private RedisCache redisCache;
    @Resource
    private TokenProperties tokenProperties;
    private static final long MILLIS_SECOND = 1000;
    private static final long MILLIS_MINUTE = 60 * MILLIS_SECOND;

    /**
     * 创建用户登录token
     *
     * @param loginUser 登录用户信息，包含用户的基本信息和认证相关信息
     * @return 返回生成的token字符串，该token用于用户认证和后续请求的验证
     */
    public String createToken(LoginUser loginUser) {
        // 生成一个唯一的UUID作为token的值，并将其保存在loginUser对象中
        String uuid = IdUtil.fastSimpleUUID();
        loginUser.setToken(uuid);
        loginUser.setOs(System.getProperty("os.name")); // 记录登录用户的操作系统信息

        // 刷新用户的token，确保其有效性
        refreshToken(loginUser);
        HashMap<String, Object> claims = new HashMap<>(); // 准备token的声明信息
        claims.put(LoginConstants.LOGIN_USER_KEY, uuid); // 将用户token作为声明之一
        try {
            // 使用JWT创建token
            return JwtUtils.createToken(claims, SignatureAlgorithm.HS256, tokenProperties.getSecret());
        } catch (Exception e) {
            // 记录创建token失败的日志
            LOGGER.error("创建Token失败", e);
            // 抛出创建token失败的异常
            throw new RuntimeException(e.getMessage());
        }
    }


    /**
     * 刷新用户登录token，更新过期时间
     *
     * @param loginUser 登录用户信息，包含用户标识和登录token等信息
     */
    public void refreshToken(LoginUser loginUser) {
        // 更新登录时间和过期时间
        loginUser.setLoginTime(new Date().getTime());
        long time = loginUser.getLoginTime();
        loginUser.setExpireTime(time + tokenProperties.getExpireTime() * MILLIS_MINUTE);
        String tokenKey = RedisCacheKey.loginToken(loginUser.getToken());
        // 在Redis中更新用户登录信息，包括token及其对应的登录用户信息，设置过期时间
        redisCache.setCacheObject(tokenKey, loginUser, tokenProperties.getExpireTime(), TimeUnit.MINUTES);
    }

    /**
     * 验证token是否有效，如即将过期则刷新token
     * 该方法会检查当前登录用户的token是否即将过期，如果将在一分钟内过期，则会刷新token
     *
     * @param loginUser 登录用户信息，包含用户的token及其过期时间等信息
     */
    public void checkToken(LoginUser loginUser) {
        Long expireTime = loginUser.getExpireTime(); // 获取登录用户的token过期时间
        long currentTime = System.currentTimeMillis(); // 获取当前时间
        // 判断如果token将在一分钟内过期，则刷新token
        if (expireTime - currentTime <= tokenProperties.getExpireTime() * MILLIS_MINUTE) {
            refreshToken(loginUser);
        }
    }

    /**
     * 从Redis中移除登录用户的信息
     * 通过给定的token找出对应的用户key，然后从Redis中删除该key，实现用户登出功能
     *
     * @param uuid 用户的登录令牌，用于标识唯一用户
     */
    public void removeLoginUser(String uuid) {
        // 根据token从Redis删除登录用户信息
        if (StringUtils.isNotEmpty(uuid)) {
            String userKey = RedisCacheKey.loginToken(uuid);
            redisCache.deleteObject(userKey);
        }
    }

    /**
     * 从请求中获取登录用户信息
     * 该方法首先从HTTP请求的头部获取token，然后验证token的有效性。如果token有效
     * 则解析token以获取登录用户的详细信息，并返回该用户对象。如果无法获取有效token
     * 或者token解析失败，则返回null
     *
     * @param request HTTP请求对象，用于获取请求头中的token
     * @return 登录用户对象，如果存在有效token；否则返回null
     */
    public LoginUser getLoginUser(HttpServletRequest request) {
        // 尝试从请求头中提取token
        String token = JwtUtils.getRequestToken(tokenProperties.getHeader(), tokenProperties.getPrefix(), request);
        if (StringUtils.isNotEmpty(token)) {
            try {
                // 验证token并解析，获取用户UUID
                Claims claims = JwtUtils.parseToken(token, tokenProperties.getSecret());
                String uuid = (String) claims.get(LoginConstants.LOGIN_USER_KEY);
                // 从缓存中获取登录用户对象
                return redisCache.getCacheObject(RedisCacheKey.loginToken(uuid),LoginUser.class);
            } catch (ServiceException e) {
                // token验证或解析失败，抛出异常
                throw new ServiceException(e.getMessage());
            }
        }
        // 未获取到有效token，返回null
        return null;
    }

    /**
     * 刷新登录用户信息
     *
     * @param loginUser 登录用户对象
     */
    public void refreshLoginUser(LoginUser loginUser) {
        // 若用户对象非空且token有效，则刷新用户登录信息
        if (StringUtils.isNotNull(loginUser) && StringUtils.isNotEmpty(loginUser.getToken())) {
            refreshToken(loginUser);
        }
    }
}

package com.silence.framework.security.service;

import com.silence.common.core.redis.RedisCache;
import com.silence.common.core.redis.RedisCacheKey;
import com.silence.common.enums.I18nLoginEnums;
import com.silence.common.exception.ServiceException;
import com.silence.common.properties.PassWordProperties;
import com.silence.framework.security.context.AuthenticationContext;
import com.silence.framework.security.context.SecurityUtils;
import org.springframework.security.core.Authentication;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;
import java.util.concurrent.TimeUnit;

/**
 * @author SilenceLambs
 * @apiNote 密码校验接口实现类
 */
@Component
public class PasswordService {
    @Resource
    private RedisCache redisCache;
    @Resource
    private PassWordProperties passWordProperties;

    /**
     * 验证提供的编码后的密码是否匹配当前用户的认证信息。
     *
     * @param encodedPassword 编码后的密码，用于与用户输入的密码进行匹配。
     * 该方法首先从认证上下文中获取当前用户的认证信息，然后对比用户输入的密码和系统存储的编码后密码。
     * 若密码不匹配或密码错误尝试次数超过预设的最大次数，则会抛出相应的异常。
     * 登录成功后，会清除该用户的密码错误尝试次数记录。
     */
    public void matches(String encodedPassword) {
        Authentication authentication = AuthenticationContext.getAuthentication();
        String inputUsername = authentication.getName(); // 获取输入的用户名
        String inputPassword = authentication.getCredentials().toString(); // 获取输入的密码
        Integer retryCount = redisCache.getCacheObject(RedisCacheKey.passwordErrorCount(inputUsername),Integer.class); // 从Redis获取当前用户密码错误尝试次数
        if (retryCount == null) {
            retryCount = 0; // 如果没有记录，则初始化为0
        }
        // 检查密码错误尝试次数是否达到锁定阈值
        if (retryCount >= passWordProperties.getMaxRetryCount()) {
            // 如果超过最大尝试次数，则抛出账户锁定异常
            int maxRetryCount = passWordProperties.getMaxRetryCount();
            int lockTime = passWordProperties.getLockTime();
            String message = I18nLoginEnums.LOGIN_USER_ACCOUNT_IS_LOCK.message(maxRetryCount, lockTime);
            throw new ServiceException(message);
        }
        // 对比输入密码和编码后的密码
        boolean matchesPassword = SecurityUtils.matchesPassword(inputPassword, encodedPassword);
        if (!matchesPassword) {
            retryCount++; // 如果密码不匹配，错误尝试次数加一
            // 将更新后的错误尝试次数缓存到Redis
            redisCache.setCacheObject(RedisCacheKey.passwordErrorCount(inputUsername), retryCount, passWordProperties.getLockTime(), TimeUnit.MINUTES);
            String message = I18nLoginEnums.LOGIN_USER_PASSWORD_NOT_MATCH.message();
            throw new ServiceException(message); // 抛出密码不匹配异常
        } else {
            // 登录成功，清除密码错误尝试次数记录
            if (redisCache.hasKey(RedisCacheKey.passwordErrorCount(inputUsername))) {
                redisCache.deleteObject(RedisCacheKey.passwordErrorCount(inputUsername));
            }
        }
    }
}

    private ServerUserService serverUserService;
    @Resource
    private ServerRoleService serverRoleService;
    @Resource
    private PasswordService passwordService;

    /**
     * 根据用户名定位用户。在实际实现中，搜索可能区分大小写或不区分大小写，具体取决于实现实例的配置方式
     *
     * @param userEmail 标识需要其数据的用户的用户名
     * @return 完全填充的用户记录
     */
    @Override
    public UserDetails loadUserByUsername(String userEmail) throws UsernameNotFoundException {
        /* 自定义查询逻辑查询用户信息 */
        ServerUser serverUser = null;
        try {
            serverUser = serverUserService.selectUserByEmail(userEmail);
        } catch (Exception e) {
            throw new ServiceException(I18nLoginEnums.LOGIN_USER_FAIL.message());
        }
        if (StringUtils.isNull(serverUser)) {
            throw new ServiceException(I18nLoginEnums.LOGIN_USER_IS_NOT_EXIST.message(userEmail));
        } else if (LoginConstants.USER_STATUS_DELETE.equals(serverUser.getDelFlag())) {
            throw new ServiceException(I18nLoginEnums.LOGIN_USER_IS_DELETED.message(userEmail));
        } else if (LoginConstants.USER_STATUS_DISABLE.equals(serverUser.getStatus())) {
            throw new ServiceException(I18nLoginEnums.LOGIN_USER_IS_DEACTIVATED.message(userEmail));
        }
        passwordService.matches(serverUser.getPassword());
        // 初始化角色键集合和菜单权限集合
        Long userId = serverUser.getUserId();
        /* 自定义查询逻辑查询用户信息 */
        List<String> roleKeyList = serverRoleService.selectRoleIdListByUserId(userId);
        // 构建并返回LoginUserDetail对象
        return new LoginUser(userId, serverUser, roleKeyList);
    }

package com.silence.common.utils;

import com.silence.common.exception.UtilException;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.net.URL;
import java.nio.charset.StandardCharsets;

/**
 * @author SilenceLambs
 * @apiNote Html工具类
 */
public class HtmlUtils {

  /**
   * 读取指定HTML资源文件的内容，并将其作为一个字符串返回。
   *
   * @param resourceName 要读取的HTML资源文件的相对路径名
   * @return 返回一个字符串，包含所读取HTML资源文件的完整内容，
   */
  public static String readHtml(String resourceName) {
    // 获取类加载器
    ClassLoader classLoader = HtmlUtils.class.getClassLoader();

    // 构建资源文件的URL
    URL resourceUrl = classLoader.getResource(resourceName);
    if (resourceUrl == null) {
      throw new UtilException("文件{}不存在 ", resourceName);
    }

    try (InputStream inputStream = resourceUrl.openStream();
         InputStreamReader inputStreamReader = new InputStreamReader(inputStream, StandardCharsets.UTF_8);
         BufferedReader reader = new BufferedReader(inputStreamReader)) {

      StringBuilder htmlContentBuilder = new StringBuilder();
      String line;
      while ((line = reader.readLine()) != null) {
        htmlContentBuilder.append(line).append(System.lineSeparator());
      }

      // 将读取到的所有行拼接成一个字符串并返回
      return htmlContentBuilder.toString();
    } catch (IOException e) {
      throw new UtilException(e.getMessage());
    }
  }
}

package com.silence.common.core.redis;

import cn.hutool.crypto.SecureUtil;
import cn.hutool.extra.servlet.ServletUtil;
import com.silence.common.constants.CacheConstants;
import com.silence.common.utils.ServletUtils;
import com.silence.common.utils.string.StringUtils;
import org.jetbrains.annotations.NotNull;

import javax.servlet.http.HttpServletRequest;

/**
 * @author SilenceLambs
 * @apiNote 生成缓存键的工具类
 */
public class RedisCacheKey {
    /**
     * 生成验证码的键值
     * 该方法基于提供的电子邮件地址生成一个唯一标识符（UUID），然后将其与一个固定的缓存键前缀结合
     * 用于在缓存中存储和检索验证码
     * 
     * @param email 电子邮箱地址，用于生成UUID
     * @return 验证码的缓存键。如果UUID为空，则键值仅包含固定的前缀
     */
    @NotNull
    public static String captchaCode(String email) {
        String UUID = createUUID(email);
        // 根据UUID生成验证码的缓存键，如果UUID为空，则键值仅包含固定前缀
        return CacheConstants.CAPTCHA_CODE_KEY + StringUtils.nvl(UUID, "");
    }
    
  
    /**
     * 根据提供的电子邮件地址创建一个UUID
     * 该UUID基于客户端的IP地址和User-Agent通过MD5加密生成，确保了唯一性和一定的安全性
     * 
     * @param email 用于生成UUID的电子邮件地址
     * @return 生成的UUID字符串
     */
    private static String createUUID(String email) {
        HttpServletRequest request = ServletUtils.getRequest();
        // 获取客户端的User-Agent信息
        String userAgent = ServletUtil.getHeader(request, "User-Agent", "utf-8");
        // 将电子邮件地址和User-Agent合并，通过MD5加密生成UUID
        return SecureUtil.md5(email + userAgent);
    }
}

     * 发送邮箱验证码
     *
     * @param email     需要发送验证码的邮箱地址
     * @param operation 进行的操作类型（例如：注册、重置密码等）
     */
    public void sendEmail(String email, String operation) {
        // 验证邮箱格式合法性

        String appName = systemConfigService.selectAppName();

        // 设置邮件主题
        String emailTopic = StringUtils.format("【{}】 {}验证码", appName.trim(), operation);

        // 读取并解析HTML邮件模板
        String readHtml = HtmlUtils.readHtml("templates/email.html");

            // 发送邮件
            MailUtil.send(email, emailTopic, text, true);

            // 邮件发送失败，抛出异常
            throw new ServiceException(I18nEnums.LOGIN_USER_EMAIL_SEND_FAILED_AGAIN.message());

    }
