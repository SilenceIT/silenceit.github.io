1.1 🌼【任务执行框架】 ​

  • 基本的任务执行框架🌳 AbstractHandler.java
java
package com.silence.quartz.handle;

import com.silence.quartz.constants.ScheduleConstants;
import com.silence.quartz.domain.JobEntity;
import io.swagger.annotations.ApiModelProperty;
import org.quartz.Job;
import org.quartz.JobDataMap;
import org.quartz.JobExecutionContext;
import org.quartz.JobExecutionException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.BeanUtils;

import java.util.Date;

/**
 * @author SilenceLambs
 * @apiNote 抽象quartz调用类，提供基本的任务执行框架
 */
public abstract class AbstractHandler implements Job {
  private static final Logger log = LoggerFactory.getLogger(AbstractHandler.class);

  @ApiModelProperty("线程本地变量，用于存储任务开始时间")
  private static ThreadLocal<Date> threadLocal = new ThreadLocal<>();

  /**
   * 执行任务的主方法
   *
   * @param context 任务执行上下文，包含任务的详细信息和执行环境
   * @throws JobExecutionException 如果任务执行过程中发生错误
   */
  @Override
  public void execute(JobExecutionContext context) throws JobExecutionException {
    log.info("开始执行:{}", context);
    JobEntity jobEntity = new JobEntity();

    // 获取合并后的任务数据映射，用于获取任务属性
    JobDataMap mergedJobDataMap = context.getMergedJobDataMap();

    // 从映射中获取任务实体，并将其属性复制到新的任务实体中
    JobEntity object = (JobEntity) mergedJobDataMap.get(ScheduleConstants.TASK_PROPERTIES);
    BeanUtils.copyProperties(object, jobEntity);

    try {
      // 执行任务前的逻辑
      before(context, jobEntity);
      // 执行具体的任务逻辑
      doExecute(context, jobEntity); // 由子类实现具体的任务执行逻辑
      // 执行任务后的逻辑
      after(context, jobEntity, null);
    } catch (Exception e) {
      // 记录任务执行异常，并执行任务后的逻辑
      log.error("任务执行异常  - ：", e);
      after(context, jobEntity, e);
    }
  }

  /**
   * 执行前的逻辑处理
   * 在具体任务执行前被调用，用于执行一些初始化或准备工作
   *
   * @param context 任务执行上下文
   * @param job     任务实体
   */
  public void before(JobExecutionContext context, JobEntity job) {
    threadLocal.set(new Date()); // 记录任务开始时间
  }

  /**
   * 执行后的逻辑处理
   * 在具体任务执行后被调用，用于执行一些清理或后续操作
   *
   * @param context 任务执行上下文
   * @param job     任务实体
   * @param exception 执行过程中抛出的异常，如果执行正常则为null
   */
  public void after(JobExecutionContext context, JobEntity job, Exception exception) {
    Date startTime = threadLocal.get(); // 获取任务开始时间
    threadLocal.remove(); // 清理线程本地变量
  }

  /**
   * 执行方法，由子类根据需求重载实现具体的任务执行逻辑
   *
   * @param context 工作执行上下文对象
   * @param job     系统计划任务实体
   * @throws Exception 执行过程中的异常
   */
  protected abstract void doExecute(JobExecutionContext context, JobEntity job) throws Exception;
}

package com.silence.quartz.handle;

import com.silence.quartz.domain.JobEntity;
import com.silence.quartz.utils.TaskInvokeUtils;
import org.quartz.JobExecutionContext;

/**
 * @author SilenceLambs
 * @apiNote 定时任务处理（允许并发执行）
 */
public class ConcurrentHandler extends AbstractHandler {
  /**
   * 执行方法，由子类重载
   *
   * @param context 工作执行上下文对象
   * @param job     系统计划任务
   * @throws Exception 执行过程中的异常
   */
  @Override
  protected void doExecute(JobExecutionContext context, JobEntity job) throws Exception {
    TaskInvokeUtils.invokeMethod(job);
  }
}

package com.silence.quartz.service.impl;

import com.baomidou.mybatisplus.core.conditions.query.QueryWrapper;
import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import com.silence.quartz.constants.ScheduleConstants;
import com.silence.quartz.domain.JobEntity;
import com.silence.quartz.exception.TaskException;
import com.silence.quartz.mapper.JobMapper;
import com.silence.quartz.service.JobService;
import com.silence.quartz.utils.ScheduleUtils;
import org.quartz.JobDataMap;
import org.quartz.JobKey;
import org.quartz.Scheduler;
import org.quartz.SchedulerException;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import javax.annotation.PostConstruct;
import javax.annotation.Resource;
import java.util.List;

/**
 * @author SilenceLambs
 * @apiNote 定时任务调度表 服务实现类
 */
@Service
public class JobServiceImpl extends ServiceImpl<JobMapper, JobEntity> implements JobService {
  @Resource
  private Scheduler scheduler;

  /**
   * 项目启动时的初始化定时器操作
   * 主要目的是确保定时任务与数据库中的配置保持同步，避免因手动修改数据库导致的定时任务未同步问题
   * 注意：不要手动修改数据库中的ID和任务组名，否则可能导致数据不一致
   *
   * @throws SchedulerException 如果操作调度程序时发生错误
   * @throws TaskException      如果处理任务时发生异常
   */
  @PostConstruct
  public void init() throws SchedulerException, TaskException {
    // 清空当前所有定时任务
    scheduler.clear();
    // 获取数据库中所有任务实体
    List<JobEntity> jobEntityList = baseMapper.selectList(null);
    // 遍历数据库中的任务实体，为每个任务创建并添加到定时器中
    for (JobEntity jobEntity : jobEntityList) {
      ScheduleUtils.createScheduleTask(scheduler, jobEntity);
    }
  }
}

/**
 * @author SilenceLambs
 * @apiNote 定时任务调度 服务实现类
 */
@Service
public class JobServiceImpl extends ServiceImpl<JobMapper, JobEntity> implements JobService {
    @Resource
    private Scheduler scheduler;

    /**
     * 立即运行指定的任务
     *
     * @param jobEntity 要运行的任务的调度信息，包含任务ID和任务组
     * @return boolean 如果任务成功触发，则返回true；如果任务不存在，则返回false
     * @throws SchedulerException 如果调度过程中发生错误
     */
    @Override
    @Transactional(rollbackFor = Exception.class)
    public boolean run(JobEntity jobEntity) throws SchedulerException {
        Long jobId = jobEntity.getJobId();
        String jobGroup = jobEntity.getJobGroup();
        QueryWrapper<JobEntity> jobQueryWrapper = new QueryWrapper<>();
        jobQueryWrapper.eq("job_id", jobId).eq("job_group", jobGroup);
        JobEntity properties = baseMapper.selectOne(jobQueryWrapper); // 通过任务ID查询任务详细信息
        System.out.println(properties);
        // 初始化任务参数
        JobDataMap dataMap = new JobDataMap();
        dataMap.put(ScheduleConstants.TASK_PROPERTIES, properties);
        JobKey jobKey = ScheduleUtils.getJobKey(jobId, jobGroup); // 获取任务的唯一标识
        // 检查任务是否存在
        if (scheduler.checkExists(jobKey)) {
            // 触发任务执行
            scheduler.triggerJob(jobKey, dataMap);
            return true;
        }
        return false;
    }

}
