1🌳【环境变量 】 ​

txt
import.meta.env.MODE: {string} 应用运行的模式
import.meta.env.BASE_URL: {string} 部署应用时的基本 URL.他由base 配置项决定
import.meta.env.PROD: {boolean} 应用是否运行在生产环境
import.meta.env.DEV: {boolean} 应用是否运行在开发环境 (永远与 import.meta.env.PROD相反)
import.meta.env.SSR: {boolean} 应用是否运行在 server 上
  • 🌳 .env 文件
shell
.env                # 所有情况下都会加载
.env.local          # 所有情况下都会加载，但会被 git 忽略
.env.[mode]         # 只在指定模式下加载
.env.[mode].local   # 只在指定模式下加载，但会被 git 忽略

type ProxyTargetList = Record<string, ProxyOptions>

/**
 * @Description  : 配置插件
 * @Author       : SilenceLamb
 * @Version      : V1.0.0
 */
import type { Plugin, PluginOption } from 'vite'
import uni from '@dcloudio/vite-plugin-uni'
import importDeps from './modules/importDeps'
import importComp from './modules/importComp'
import importUnocss from './modules/importUnocss'
import importStyle from './modules/importStyle'
import SvgIcons from './modules/svgIcons'
import GZIPCompression from './modules/compression'
export default function createVitePlugins(ViteEnv: any, isBuild: boolean) {
  const vitePlugins: (Plugin | Plugin[] | PluginOption)[] = []
  // 自动按需引入依赖
  vitePlugins.push(importDeps())
  // 自动引入组件
  vitePlugins.push(importComp())
  // 自定义图标
  vitePlugins.push(SvgIcons())
  // 自动引入样式
  vitePlugins.push(importStyle())
  // 美化组件样式
  vitePlugins.push(importUnocss())
  // GIZP压缩
  vitePlugins.push(GZIPCompression(isBuild))
  vitePlugins.push(uni())
  return vitePlugins
}

9🌼【Pinia 持久化】 ​

注意事项

说明：Pinia 用法与 Vue3 项目完全一致，uni-app 项目仅需解决持久化插件兼容性问题。

sh

注意事项

插件默认使用 localStorage 实现持久化，小程序端不兼容，需要替换持久化 API。

/**

 * @Author       : SilenceLamb
 * @Version      : V1.0.0
 * @Date         : 2023-09-12 04:38:50
 */
import { defineStore } from 'pinia'
import { ADMIN_INFO } from '../constant/cacheKey'
import type { AdminInfo } from '../constant/interace'
import storeApi from '@/services/store/indes'
// 定义 Store
export const useAdminInfoState = defineStore(
  {
    id: ADMIN_INFO,
    state: (): AdminInfo => {
      return {
        store: {},
        cart: [],
        orderType: 'takein',
        address: {},
        addresses: [],
        member: {},
        order: {},
        token: '',
      }
    },
    getters: {
      isLogin: (state) => Object.keys(state.member).length > 0, //是否登录
      /**
       * @Description : 获取店铺详情
       */
    },
    actions: {
      /**
       * @Description :设置state全部值
       * @param state
       */
      setState(state: AdminInfo) {
        this.$state = { ...this.$state, ...state }
      },

      /**
       * @Description : 通过键设置值
       * @param key state的键
       * @param value 值
       */
      setStateByKey(key: keyof typeof this.$state, value: any) {
        this[key] = value
      },

      /**
       * @Description : 通过键获取值
       * @param key state的键
       */
      getStateByKey(key: keyof typeof this.$state): any {
        return this[key]
      },
      async getOrderStore() {
        const response = await storeApi.DBStoreList()
        this.store = response.result.data.pop()
        return this.store
      },
    },
  },
  // TODO: 持久化
  {
    persist: true,
  },
)

10🌷 自动注册组件 ​

  • 安装 unplugin-auto-import 插件
json
pnpm add unplugin-auto-import
  • 项目根目录创建 typings 目录，然后按如下方法配置 vite.config.ts
ts
import { defineConfig } from "vite"
import uni from "@dcloudio/vite-plugin-uni"
// 加上下面这一行
import AutoImport from 'unplugin-auto-import/vite'

export default defineConfig({
  plugins: [
    uni()，
     AutoImport({
      include: [
        /\.[tj]sx?$/, // .ts, .tsx, .js, .jsx
        /\.vue$/,
        /\.vue\?vue/, // .vue
      ],
      imports: ['vue', 'uni-app'],
      dts: 'src/types/auto-imports.d.ts',
    }),
  ],
})
  • 同时在 tsconfig.json 中添加如下配置
json
"include": [
    "src/**/*.ts",
    "src/**/*.d.ts",
    "src/**/*.tsx",
    "src/**/*.vue",
    "types/**/*.ts",
    "types/**/*.d.ts"
  ]
